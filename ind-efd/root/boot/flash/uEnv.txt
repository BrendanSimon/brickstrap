kernel_load_address=0x3000000
kernel_image=/boot/uImage
devicetree_image=/boot/ind.dtb
devicetree_load_address=0x2a00000
boot_part=2
bootargs2=console=ttyPS0,115200 root=/dev/mmcblk0p2 ro earlyprintk rootfstype=ext4 rootwait
bootargs3=console=ttyPS0,115200 root=/dev/mmcblk0p3 ro earlyprintk rootfstype=ext4 rootwait
bootargsx=console=ttyPS0,115200 root=/dev/mmcblk0p${boot_part} ro earlyprintk rootfstype=ext4 rootwait
load_kernel=echo "Loading kernel ..." ; ext2load mmc 0:${boot_part} ${kernel_load_address} ${kernel_image}
load_devicetree=echo "Loading devicetree ..." ; ext2load mmc 0:${boot_part} ${devicetree_load_address} ${devicetree_image}
set_bootargs=echo "Setting bootargs ..." ; if test "${boot_part}" = "3" ; then echo "Setting bootargs3" ; setenv bootargs ${bootargs3} ; else echo "Setting bootargs2" ; setenv bootargs ${bootargs2} ; fi
boot_uimage=echo "Booting uImage ..." ; bootm ${kernel_load_address} - ${devicetree_load_address}
sdboot=if mmcinfo; then echo Copying Linux from SD to RAM... && run load_kernel && run load_devicetree && run set_bootargs && run boot_uimage ; fi
mode_boot=sdboot
uenvcmd=boot
